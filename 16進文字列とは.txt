・16進文字列とは
アスキーコード（10進文字列）を16進数で表現したもの

#include <stdio.h>

int main()
{
    printf("文字列");
    printf("%c", 67);
    printf("%c", 65);
    printf("%c ", 84);

    printf("10進数アスキー");
    printf("%d", 'C');
    printf("%d", 'A');
    printf("%d ", 'T');

    printf("16進数アスキー");
    printf("%x", 'C');
    printf("%x", 'A');
    printf("%x ", 'T');

    return 0;
}

を実行すると、

文字列CAT 10進数アスキー676584 16進数アスキー434154 

と表示される
